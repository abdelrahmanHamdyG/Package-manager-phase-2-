CREATE TABLE package (
	P_ID SERIAL PRIMARY KEY,
	Name VARCHAR NOT NULL,
	Github_Url VARCHAR NOT NULL
);

CREATE TABLE pack_version (
	Version VARCHAR,
	P_ID SERIAL,
	Correctness numeric NOT NULL DEFAULT 0,
	Responsiveness numeric NOT NULL DEFAULT 0,
	Ramp_up numeric NOT NULL DEFAULT 0,
	Bus_factor numeric NOT NULL DEFAULT 0,
	License_metric numeric NOT NULL DEFAULT 0,
	CONSTRAINT pk_package PRIMARY KEY (Version, P_ID),
	CONSTRAINT fk_package FOREIGN KEY (P_ID)
	REFERENCES package (P_ID)
);
